dars: SUM, AVG, MAX, MIN
Endi group by ni tugatamiz — yana 4 ta funksiyani o'rganasiz:

sql
-- Har bir foydalanuvchining jami sarflagan puli
SELECT users.name, SUM(orders.price) AS total_spent
FROM users
LEFT JOIN orders ON users.id = orders.user_id
GROUP BY users.name;

-- O'rtacha buyurtma narxi
SELECT users.name, AVG(orders.price) AS avg_price
FROM users
LEFT JOIN orders ON users.id = orders.user_id
GROUP BY users.name;

-- Eng qimmat buyurtma
SELECT users.name, MAX(orders.price) AS max_price
FROM users
LEFT JOIN orders ON users.id = orders.user_id
GROUP BY users.name;

-- Eng arzon buyurtma
SELECT users.name, MIN(orders.price) AS min_price
FROM users
LEFT JOIN orders ON users.id = orders.user_id
GROUP BY users.name;

	id SERIAL PRIMARY KEY,
	name VARCHAR(50),
	age INTEGER
);


CREATE TABLE orders(
	id SERIAL PRIMARY KEY,
	user_id  INTEGER REFERENCES users(id),
	product VARCHAR(100),
	price INTEGER 
);
